# Write your own code
csv_names = ['data_quiz/wdi_agedist_1980.csv', 'data_quiz/wdi_agedist_2000.csv', 'data_quiz/wdi_agedist_2020.csv']
dataset_years = [1980, 2000, 2020]

index = 0
for index in range(len(csv_names)):
    dataset = pd.read_csv(csv_names[index])
    percent_ages0to14 = dataset["percent_ages0to14"]
    proportion15orolder = 100 -  percent_ages0to14
    plt.hist(proportion15orolder)
    plt.xlabel("Proportion of people older than 15")
    plt.ylabel("Frequency")
    plt.title(dataset_years[index])
    plt.show()
    index = index + 1
